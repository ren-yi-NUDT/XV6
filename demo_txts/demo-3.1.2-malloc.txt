$48 = (Header *) 0x4000

(gdb) p/x freep->s.size
$49 = 0x2001
  //节点1的内存范围变成了[0x4000,0x24010=0x4000+0x20010)，已经合并了新释放的节点。

(gdb) p freep->s.ptr
$50 = (union header *) 0x1000 <base>

(gdb) p freep->s.ptr->s.size
$51 = 0
  //节点2是base

(gdb) p freep->s.ptr->s.ptr
$52 = (union header *) 0x4000
  //节点2的后继是节点1，释放了所有已分配内存后，空闲块链表中只有一个块，其大小是0x20010字节；
    //第1次malloc时向内核申请了0x10000字节（这是每次增加的最小值），第2次malloc时向内核申请了
    //0x10010字节（含0x10字节的块头），两次共申请了0x20010字节，现在都已空闲。
