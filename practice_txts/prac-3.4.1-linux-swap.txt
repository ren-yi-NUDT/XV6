//---文件swaptest.c的内容：
#include <stdio.h>
#include <stdlib.h>

#define NUM_OF_PAGES 4096

int main()
{
	char *buf;
	int i;
	buf = malloc(4096 * NUM_OF_PAGES);
	if (buf == 0) {
		printf("Error in malloc()\n");
		exit(1);
	}
	for (i=0; i<NUM_OF_PAGES; i++)
		buf[4096*i] = 1;
	free(buf);	
	return 0;
}
